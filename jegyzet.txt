

ssh-keygen -f "/home/molnarsg/.ssh/known_hosts" -R "k8s-mst-1"


apt install -y sudo joe mc curl wget gnupg software-properties-common


edit /etc/fstab, remove or comment swap line(s)
swapoff -a

curl -fsSL https://download.docker.com/linux/debian/gpg | apt-key add -

add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/debian $(lsb_release -cs) stable"

apt-get update && apt-get install -y docker.io && apt-mark hold docker.io

systemctl enable docker.service


cat > /etc/docker/daemon.json <<EOF
{
  "exec-opts": ["native.cgroupdriver=systemd"],
  "log-driver": "json-file",
  "log-opts": {
    "max-size": "100m"
  },
  "storage-driver": "overlay2"
}
EOF

mkdir -p /etc/systemd/system/docker.service.d


systemctl daemon-reload && systemctl restart docker

curl -s https://packages.cloud.google.com/apt/doc/apt-key.gpg | sudo apt-key add -
